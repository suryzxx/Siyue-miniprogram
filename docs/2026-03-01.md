小程序手机号登录
```javascript
const axios = require('axios');
let data = JSON.stringify({
  "code": ""
});

let config = {
  method: 'post',
  maxBodyLength: Infinity,
  url: 'localhost:30001/client/open/user/miniapp/login',
  headers: { 
    'Authorization': '2bd733551dfd415b9e70644c9cf19ddf', 
    'Content-Type': 'application/json'
  },
  data : data
};

axios.request(config)
.then((response) => {
  console.log(JSON.stringify(response.data));
})
.catch((error) => {
  console.log(error);
});
```

//响应

```json
{
        "code": 200,
        "data": {
                "token": "token",
                "is_temp_token": true //如果为true说明第一次注册，要跳转到新建学生页面
        }
}
```
